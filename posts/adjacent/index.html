隣接行列と隣接リストの変換

under algorithm

グラフで使われる、隣接行列から隣接リストへの変換関数を作る。

alist **listify2(int n,int mat[n][n], int boundary){
    int i,j;
    alist* new;
    alist** ls;
    for(i=0;i<n;i++){
        ls[i] = malloc(sizeof(alist));
        ls[i]->idx = -1;
        ls[i]->next = ls[i];
    }

    for(i = 0; i < n; i++){
        for(j = 0; j < n; j++){
            if(mat[i][j] != boundary){
                // insert cost and key
                new = (alist *)malloc(sizeof(alist));
                new->cost = mat[i][j];
                new->idx = j;
                // pointer replace
                new->next = ls[i]->next;
                ls[i]->next = new;
                
            }
        }
    }
    return ls;

}

#define N 8
#define M 10000
int main(void){
    
    int edge[N][N]= {{0,1,7,2,M,M,M,M},
                {1,0,M,M,2,4,M,M},
                {7,M,0,M,M,2,3,M},
                {2,M,M,0,M,M,5,M},
                {M,2,M,M,0,1,M,M},
                {M,4,2,M,1,0,M,6},
                {M,M,3,5,M,M,0,2},
                {M,M,M,M,M,6,2,0}};

    alist **lst2 = listify2(N,edge,M);
    printf("%p\n",lst2);
    printf("%p\n",lst2[0]);
    printf("%p\n",lst2[1]);

    printf("%d\n",lst2[2]->next->idx);
}
