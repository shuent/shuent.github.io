最小全域木 プリム法とクラスカル法

under algorithm

最小全域木を求める二つのアルゴリズム、プリム法とクラスカル法をC言語で実装するよ。クラスカル法は、Union-findを使う。

最小全域木とは

グラフ G(V,E) のすべての頂点と、閉路を持たない辺だけのこしたものを全域木と呼ぶ。そのなかでも、全域木の中で、各辺の総和が最小になるような全域木を最小全域木という。

プリム法

ダイクストラに似ている。疑似コードや計算量の解説はWikipediaがわかりやすい。プリム法

#include <stdio.h>
#include <stdlib.h>

// 隣接リストとで辺が表現されていると仮定。
// size はつながっているノードの数
typedef struct Node {
    int idx; int cost;
    struct Node *next;
} Node;

// 隣接リストにつなげる。
Node *connect(Node **lst, int from, int to, int cost){

    Node *x = (Node *)malloc(sizeof(Node));
    x->cost = cost;
    x->idx = to;
    x->next = lst[from]->next;
    lst[from]->next = x;
}

int main(){
    int i,j,k,t,u,min;
    int v_size = 5;
    Node *adj[v_size]; // adjacent list
    Node *newn,*p;
    int parent[v_size]; //costは取得しない

    // 適当に隣接リストを埋める。
    // 無向グラフにするため、辺は対称にする。
    for(i=0;i<v_size;i++){
        adj[i] = (Node *)malloc(sizeof(Node));
        adj[i]->idx = -1;
        adj[i]->next = adj[i];
    }
    connect(adj,0,1,5);
    connect(adj,1,0,5);
    connect(adj,0,2,34);
    connect(adj,2,0,34);
    connect(adj,2,4,8);
    connect(adj,4,2,8);
    connect(adj,4,3,20);
    connect(adj,3,4,20);
    connect(adj,0,4,14);
    connect(adj,4,0,14);
    connect(adj,1,3,26);
    connect(adj,3,1,26);


    // 確認
    
    for(i=0;i<v_size;i++){
        p = adj[i]->next;
        while(p!=adj[i]){
            printf("%d:%d ",p->idx,p->cost);
            p = p->next;
        }
        putchar('\n');
    }

    // 本体
    
    int ref;
    int visited[v_size];
    int cost[v_size];
    const int INF = 1<<20;
    for(i=0;i<v_size;i++){
        cost[i]=INF;
        visited[i]=0;
    }
    cost[0]=0;
    for(i=0;i<v_size;i++){
        min = INF;
        for(j=0;j<v_size;j++){
            if(!visited[j] && min > cost[j]){
                ref = j;
                min = cost[j];
            }
        }

        if(min == INF) exit(0); //連結でないので終了
        visited[ref] = 1;
        
        p = adj[ref]->next;
        while(p!=adj[ref]){
            if(!visited[p->idx] && cost[ref] + p->cost < cost[p->idx] ){
                cost[p->idx] = cost[ref] + p->cost;
                parent[p->idx] = ref;
            }
            p = p->next;
        }
    }
    for(i=0;i<v_size;i++){
        printf("%d->%d ",i,parent[i]);
    }
    putchar('\n');
    for(i=0;i<v_size;i++){
        printf("%d ",cost[i]);
    }
    putchar('\n');
}

ここで俺は重大発見に気づいてしまったかも知れなかった。隣接リストをそれぞれコスト順に並べれば、O(1)で最小の繋がっているノードを取得できることになる。

Union-find

アルゴリズムの中で、同じ集合に点が属するか判定する必要があるため、素集合データ構造を使う。その操作にUnion-findアルゴリズムを使う。

#include <stdio.h>
#include <stdlib.h>

typedef struct node{
    int key;
    struct node* par;
} node;

node *makeSet(int x){
    node *n = (node *)malloc(sizeof(node));
    n->key = x; n->par = n;
    return n;
}

node *findRoot(node *x){
    if (x->par->key == x->key) return x->par;
    else {
        node **p = &x;
        (*p)->par = findRoot(x->par);
    }
}

void unite(node *x, node *y){
    node *p,*q;
    p = findRoot(x); q = findRoot(y);
    p->par = q->par;
}

int is_same(node *x, node *y){
    return findRoot(x) == findRoot(y);
}


int test(){
    const int N=5;
    int i;
    node *nodes[N];
    for(i=0;i<N;i++)
        nodes[i] = makeSet(i);
    
    // find root test
    nodes[0]->par = nodes[1];
    node * p = findRoot(nodes[0]);
    printf("%p \n", p);
    printf("%p \n", nodes[1]->par);
    
    // unite test
    unite(nodes[0],nodes[1]);
    unite(nodes[2],nodes[3]);
    for(i=0;i<N-1;i++)
        printf("is same %d\n", is_same(nodes[i],nodes[i+1]));     

}

クラスカル

アルゴリズムを簡単に説明する。

  • 頂点をそれぞれ要素が一つだけの素な集合とおく。
  • 集合Eにすべての辺を入れ、コスト昇順にソートする、もしくはpriority queueにいれる。
  • 最小のコストを持つ辺をEから取り出し、始点と終点の属する集合が互いに素な場合、併合する。この工程辺がなくなるまで繰り返す。

辺は、始点　終点　コスト　という行がE個与えられると仮定。隣接リストや隣接行列でグラフが与えられた場合、最小の辺を見つけて並べ替えるのに時間がかかるので、その場合はプリムを使うといいかも。

辺のソートには、C言語の標準ライブラリの qsort()を利用した。

#include "union-find.c"
#define E_MAX 10000
#define V_MAX 100

typedef struct {
    int from; int to; int cost;
} Edge;
// Edge *edges[E_MAX];
// Edge *min_edges[V_MAX];
// node *vertex[V_MAX];

// qsort用の比較関数
int comp(const void *a, const void *b){
    Edge **p = (Edge **)a;
    Edge **q = (Edge **)b;
    return (*p)->cost - (*q)->cost;
}

int main(){
    int i,head=0;
    int v_size = 5;    
    int e_size = 15;
    int me_size=0;

    Edge *min_edges[v_size];
    node *vertex[e_size];

    // 適当に辺を設定する
    Edge *edges[e_size];
    for(i=0;i<e_size;i++){
        edges[i] = (Edge *)malloc(sizeof edges[0]);
        edges[i]->from = i % v_size;
        edges[i]->to = (i*71+ rand()%237)% v_size;
        // edges[i]->to = i%v_size;
        edges[i]->cost = ((i + i*3+ 5)*(i+7)*7 + 11)%23;
    }
    
    // for(i=0;i<e_size;i++)
    //     printf("%d->%d, cost %d\n", edges[i]->from, 
    //                                 edges[i]->to, 
    //                                 edges[i]->cost);

    // ソートする。
    printf("\n");
    qsort(edges, e_size, sizeof(Edge *), comp);

    // ソートされたか確認
    for(i=0;i<e_size;i++)
        printf("%d->%d, cost %d\n", edges[i]->from, 
                                    edges[i]->to, 
                                    edges[i]->cost);

    //それぞれ頂点を素な集合にする
    for(i=0;i<v_size;i++){
        vertex[i] = makeSet(i);
        // printf("%d ",vertex[i]->key);
    }
    
    // while (me_size < v_size-1){　// 連結木でない場合終了しない
    for(i=0;i< e_size;i++){ // 辺の数以下。
        Edge *e = edges[head++];
        if (!is_same( vertex[e->from], vertex[e->to])){
            unite(vertex[e->from], vertex[e->to]);
            // printf("%d %d\n",vertex[e->from]->key,vertex[e->to]->key);
            min_edges[me_size++] = e;
        }
    }

    // 最小全域木の辺をすべて表示
    for(i=0;i<me_size;i++)
        printf("%d->%d, cost %d\n", min_edges[i]->from, 
                                    min_edges[i]->to, 
                                    min_edges[i]->cost);
    
}
